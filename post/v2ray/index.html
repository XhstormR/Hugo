服务端

docker-compose.yml

version: '3'

services:
  v2ray:
    image: v2fly/v2fly-core:latest
    restart: always
    ports:
      - '8080:8080'
    volumes:
      - ./config.json:/etc/v2ray/config.json:ro
      - ./tls_key.pem:/root/tls_key.pem:ro
      - ./tls_cert.pem:/root/tls_cert.pem:ro
